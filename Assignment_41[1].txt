#include <stdio.h>

int main() {
    int age, choice;

    printf("Enter your age: ");
    scanf("%d", &age);

    // Using switch case
    switch (age >= 18) {
        case 1:
            printf("You are eligible to vote.\n");
            break;
        case 0:
            printf("You are not eligible to vote.\n");
            break;
        default:
            printf("Invalid input.\n");
    }

    return 0;
}
