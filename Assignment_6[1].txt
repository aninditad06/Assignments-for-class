#include <stdio.h>

int main() {
    float principal, rate, time, si;

    // Input values
    printf("Enter Principal amount: ");
    scanf("%f", &principal);

    printf("Enter Rate of Interest: ");
    scanf("%f", &rate);

    printf("Enter Time (in years): ");
    scanf("%f", &time);

    // Formula: S.I = (P × R × T) / 100
    si = (principal * rate * time) / 100;

    printf("Simple Interest = %.2f\n", si);

    return 0;
}
