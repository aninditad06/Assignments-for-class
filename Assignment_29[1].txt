#include <stdio.h>
int main() {
    int units;
    float bill;
    printf("Enter units consumed: ");
    scanf("%d", &units);
    switch(units / 100) {
        case 0: bill = units * 1.0; break;
        case 1: bill = units * 1.5; break;
        case 2: bill = units * 2.0; break;
        default: bill = units * 2.5; break;
    }
    printf("Electricity bill = %.2f\n", bill);
    return 0;
}
